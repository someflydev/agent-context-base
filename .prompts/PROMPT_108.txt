  class TerminalExample:
    name: str
    path: Path                   # e.g., examples/canonical-terminal/python-typer-textual
    build_cmd: list[str]         # command to build (empty if not needed before testing)
    smoke_cmd: list[str]         # command to run the CLI list subcommand (no-tui mode)

  def run_smoke(example: TerminalExample) -> SmokeResult:
    """
    Run the smoke command for an example.
    Returns SmokeResult with: exit_code, stdout, stderr, duration_s.
    Uses subprocess.run with timeout=30, no TTY (no stdin).
    Sets TASKFLOW_FIXTURES_PATH env var.
    """

  class SmokeResult:
    exit_code: int
    stdout: str
    stderr: str
    duration_s: float
    passed: bool  # exit_code == 0 and expected_marker in stdout

  def run_all(examples: list[TerminalExample], parallel=False) -> dict[str, SmokeResult]:
    """Run smoke tests for all examples. Optional parallel execution."""

Create: verification/terminal/transcript.py

Transcript normalization strips non-deterministic elements from CLI output
to enable golden/snapshot testing.

  def normalize_transcript(text: str) -> str:
    """
    Normalize CLI output for golden comparison:
    - Replace timestamps (ISO8601 patterns) with <TIMESTAMP>
    - Replace UUIDs / job IDs like job-001 with <JOB_ID> (optional, configurable)
    - Replace duration_s values like 47.3 with <DURATION>
    - Strip ANSI escape codes
    - Normalize whitespace (multiple spaces → single space, trailing whitespace)
    - Keep markers (## BEGIN_JOBS ## etc.) intact
    Returns normalized string.
    """

  def diff_transcripts(expected: str, actual: str) -> str:
    """Return unified diff of two normalized transcripts."""

  def save_golden(name: str, content: str, golden_dir: Path) -> None:
    """Save normalized transcript as golden file."""

  def load_golden(name: str, golden_dir: Path) -> str | None:
    """Load golden transcript for comparison. Returns None if not yet created."""

  def assert_transcript(name: str, actual: str, golden_dir: Path,
                        update=False) -> bool:
    """
    Compare actual (normalized) against golden.
    If update=True: overwrite golden (used to regenerate expected output).
    If golden doesn't exist: save it and return True (first-run bootstrap).
    If differs: return False and print diff.
    """

Create: verification/terminal/pty_harness.py

PTY harness for scripted interaction with TUI applications. Uses pexpect.

  import pexpect

  class PtySession:
    """
    Wraps a pexpect.spawn session for scripted TUI interaction.
    """
    def __init__(self, cmd: list[str], env: dict = None, timeout: int = 10):
      self.child = pexpect.spawn(cmd[0], cmd[1:], env=env, timeout=timeout,
                                  encoding='utf-8', codec_errors='replace')

    def expect(self, pattern: str, timeout: int = None) -> str:
      """Wait for pattern in output. Returns matched text."""
      self.child.expect(pattern, timeout=timeout)
      return self.child.before + self.child.after

    def send_key(self, key: str) -> None:
      """Send a key sequence (e.g., 'q', '\r', pexpect.CTRL_C)."""
      self.child.send(key)

    def send_line(self, line: str) -> None:
      self.child.sendline(line)

    def close(self) -> int:
      """Close session. Returns exit code."""
      self.child.close()
      return self.child.exitstatus

  def scripted_tui_test(cmd: list[str], script: list[dict], env: dict = None) -> bool:
    """
    Run a TUI app with a scripted interaction sequence.
    script: list of {action: "expect"|"send_key", value: str}
    Returns True if all steps succeed and app exits cleanly.

    Example script:
      [
        {"action": "expect", "value": "BEGIN_JOBS|job-001"},  # wait for TUI to load
        {"action": "send_key", "value": "q"},                  # quit
        {"action": "expect_exit", "value": 0}                  # assert exit code 0
      ]
    """

  def make_watch_script() -> list[dict]:
    """
    Standard script for testing 'taskflow watch' TUI:
    1. Expect job list to appear (wait for a job name from fixtures)
    2. Press 'q' to quit
    3. Expect exit code 0
    """
    return [
      {"action": "expect", "value": "build-frontend|job-001|TaskFlow"},
      {"action": "send_key", "value": "q"},
      {"action": "expect_exit", "value": 0}
    ]
