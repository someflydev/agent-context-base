PART 2: EVENT-STREAM REPLAY MODULE
================================================================================

Create: examples/canonical-terminal/fixtures/replay/

This provides event-stream replay: simulate a live job queue by replaying events
from events.json in chronological order with configurable speed.

--- replay/stream.py (Python) ---

  import json, time
  from pathlib import Path
  from typing import Iterator

  def load_events(events_path: Path) -> list[dict]:
    with open(events_path) as f:
      return json.load(f)

  def sort_by_timestamp(events: list[dict]) -> list[dict]:
    return sorted(events, key=lambda e: e["timestamp"])

  def replay_events(events_path: Path, speed_factor: float = 1.0,
                    max_events: int = None) -> Iterator[dict]:
    """
    Generator. Yields events in timestamp order.
    Sleeps between events proportional to real time gaps, scaled by speed_factor.
    speed_factor=1.0: real time. speed_factor=10.0: 10x faster. speed_factor=0.0: no sleep.
    max_events: stop after N events.
    """
    events = sort_by_timestamp(load_events(events_path))
    if max_events:
      events = events[:max_events]
    prev_ts = None
    for event in events:
      if prev_ts and speed_factor > 0:
        delta = parse_delta(event["timestamp"], prev_ts)
        time.sleep(delta / speed_factor)
      yield event
      prev_ts = event["timestamp"]

  def replay_to_state(events_path: Path, jobs_path: Path,
                      up_to_event: int = None) -> list[dict]:
    """
    Apply events to job list up to the Nth event.
    Returns the resulting job list state.
    Useful for snapshot testing without sleep.
    """

  def parse_delta(ts_new: str, ts_old: str) -> float:
    """Return seconds between two ISO8601 timestamps."""

--- replay/README.md ---

  # Event Stream Replay

  Simulates a live job queue by replaying fixture events in timestamp order.

  ## Python
  ```python
  from replay.stream import replay_events

  for event in replay_events(events_path, speed_factor=10.0):
      print(f"[{event['event_type']}] {event['job_id']}: {event['message']}")
  ```

  ## Go
  ```go
  replayer := replay.NewReplayer(eventsPath, replay.SpeedFactor(10.0))
  for replayer.Next() {
      event := replayer.Event()
      // update job state
  }
  ```

  ## Use in Terminal Examples
  Pass --live-replay flag to watch commands to simulate live data.
  All replay is fixture-backed: no network required.

  ## Snapshot Mode
  Use replay_to_state(events_path, jobs_path, up_to_event=N) to compute
  the job list state at a specific point in the event stream, without sleep.
  Useful for deterministic snapshot tests.

  Tests that ensure the fixture corpus is always coherent:

  class TestFixtureCorpus(unittest.TestCase):

    def setUp(self):
      self.fixtures_dir = REPO_ROOT / "examples/canonical-terminal/fixtures"
      with open(self.fixtures_dir / "jobs.json") as f:
        self.jobs = json.load(f)
      with open(self.fixtures_dir / "events.json") as f:
        self.events = json.load(f)
      with open(self.fixtures_dir / "jobs-large.json") as f:
        self.jobs_large = json.load(f)

    def test_jobs_count(self):
      self.assertEqual(len(self.jobs), 20)

    def test_jobs_large_count(self):
      self.assertEqual(len(self.jobs_large), 100)

    def test_events_count(self):
      self.assertEqual(len(self.events), 30)

    def test_jobs_have_required_fields(self):
      required = {"id", "name", "status", "tags", "output_lines"}
      for job in self.jobs:
        self.assertTrue(required.issubset(job.keys()), f"job {job.get('id')} missing fields")

    def test_job_statuses_valid(self):
      valid = {"pending", "running", "done", "failed"}
      for job in self.jobs:
        self.assertIn(job["status"], valid)

    def test_jobs_have_unique_ids(self):
      ids = [j["id"] for j in self.jobs]
      self.assertEqual(len(ids), len(set(ids)))

    def test_jobs_large_have_unique_ids(self):
      ids = [j["id"] for j in self.jobs_large]
      self.assertEqual(len(ids), len(set(ids)))

    def test_events_reference_valid_jobs(self):
      job_ids = {j["id"] for j in self.jobs}
      for evt in self.events:
        self.assertIn(evt["job_id"], job_ids,
                      f"event {evt.get('event_id')} references unknown job {evt.get('job_id')}")

    def test_done_jobs_have_duration(self):
      for job in self.jobs:
        if job["status"] == "done":
          self.assertIsNotNone(job.get("duration_s"),
                               f"done job {job['id']} missing duration_s")

    def test_pending_jobs_have_no_started_at(self):
      for job in self.jobs:
        if job["status"] == "pending":
          self.assertIsNone(job.get("started_at"),
                            f"pending job {job['id']} should not have started_at")

    def test_config_json_exists_and_valid(self):
      config_path = self.fixtures_dir / "config.json"
      self.assertTrue(config_path.exists())
      with open(config_path) as f:
        config = json.load(f)
      self.assertIn("queue_name", config)

    def test_edge_cases_file_exists(self):
      edge_path = self.fixtures_dir / "fixtures-edge-cases.json"
      self.assertTrue(edge_path.exists())
      with open(edge_path) as f:
        edges = json.load(f)
      self.assertGreaterEqual(len(edges), 5)
