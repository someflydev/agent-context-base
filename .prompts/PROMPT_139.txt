BACKGROUND: RUBY AND ELIXIR AUTH PATTERNS
================================================================================

Ruby (Hanami + ruby-jwt):
  - ruby-jwt is the canonical JWT library for Ruby. It is the cryptographic
    base that warden-jwt_auth and devise-jwt build on top of. For this
    canonical example, use ruby-jwt directly — no Devise, no Warden.
  - Hanami uses Rack under the hood. JWT validation is implemented as a
    Rack middleware class.
  - Auth context is passed via the Rack env hash. Hanami actions can access
    it through the request object.
  - ruby-jwt v2.x API: JWT.decode(token, key, true, algorithms: ["RS256"],
    iss: "tenantcore-auth", verify_iss: true, aud: "tenantcore-api",
    verify_aud: true, exp_leeway: 0)
  - The decode call raises JWT::DecodeError and subclasses on failure.
  - RBAC: Hanami supports before-action hooks — use a before :authenticate!
    and before :require_permission!(<permission>) pattern.

Elixir (Phoenix + Joken):
  - Joken is the idiomatic Elixir JWT library. It wraps JOSE/erlang-jose
    internally but provides a much cleaner Elixir-native API.
  - A Phoenix Plug pipeline is the canonical auth middleware point.
  - Auth context is stored in conn.assigns.
  - Joken configuration: use Joken.Config and define a token_config/0 that
    specifies claim validators.
  - JOSE-backed RS256 signing: JOSE.JWK.from_pem(key_pem) → signer.
  - For the canonical test mode: use HS256 with a test secret set via config.
  - RBAC: Phoenix Plugs as permission guards:
      plug :require_permission, "iam:user:read" when action in [:index, :show]
  - This is NOT a LiveView example. Phoenix controllers + HTMX is the pattern.
    Refer to context/stacks/elixir-phoenix-joken-rbac.md for the rationale.

  lib/tenantcore_auth/auth/jwt_middleware.rb:
    class JwtMiddleware
      AUTH_CONTEXT_KEY = "tenantcore.auth_context".freeze

      def initialize(app, store:, verify_key:)
        @app = app; @store = store; @verify_key = verify_key
      end

      def call(env)
        token = extract_token(env)
        if token.nil?
          env[AUTH_CONTEXT_KEY] = nil
          return @app.call(env)
        end

        begin
          payload, _header = JWT.decode(
            token, @verify_key, true,
            algorithms: ["RS256"],
            iss: "tenantcore-auth", verify_iss: true,
            aud: "tenantcore-api", verify_aud: true
          )
        rescue JWT::DecodeError
          return [401, {"Content-Type" => "application/json"}, ['{"error":"Unauthorized"}']]
        end

        user = @store.get_user_by_id(payload["sub"])
        return [401, ...] unless user
        return [401, ...] if payload["acl_ver"] != user.acl_ver

        tenant_id = payload["tenant_id"]
        if tenant_id && !@store.verify_membership(user.id, tenant_id)
          return [403, {"Content-Type" => "application/json"}, ['{"error":"Forbidden"}']]
        end

        env[AUTH_CONTEXT_KEY] = AuthContext.new(
          sub: payload["sub"],
          email: user.email,
          tenant_id: tenant_id,
          tenant_role: payload["tenant_role"],
          groups: payload["groups"] || [],
          permissions: payload["permissions"] || [],
          acl_ver: payload["acl_ver"],
          issued_at: Time.at(payload["iat"]),
          expires_at: Time.at(payload["exp"])
        )
        @app.call(env)
      end

      private

      def extract_token(env)
        auth = env["HTTP_AUTHORIZATION"]
        auth&.start_with?("Bearer ") ? auth.sub("Bearer ", "").strip : nil
      end
    end

  RBAC pattern in Hanami actions:
    In a base authenticated action:
      before :authenticate!

      def authenticate!
        @auth = request.env["tenantcore.auth_context"]
        halt 401, headers, '{"error":"Unauthorized"}' unless @auth
      end

    For permission-specific actions, use a parameterized before hook:
      before :require_permission_iam_user_read
      def require_permission_iam_user_read
        halt 403 unless @auth&.has_permission?("iam:user:read")
      end

    Note: Hanami before hooks do not support parameters in all versions.
    If using Hanami 2.x, use a helper module pattern. Read the Hanami
    version installed (Gemfile) and adapt accordingly.

    Alternative for simpler Rack-level implementations:
    Check permission at the top of each action's handle method:
      def handle(request, response)
        authenticate!(request)
        require_permission!(request, "iam:user:read")
        # ... handler logic
      end

    class TokenService
      def initialize(signing_key:)
        @signing_key = signing_key
      end

      def issue_token(user, store)
        permissions = store.get_effective_permissions(user.id)
        groups = store.get_groups_for_user(user.id, user.tenant_id).map(&:slug)
        tenant_role = store.get_membership_role(user.id, user.tenant_id)
        now = Time.now.to_i

        payload = {
          iss: "tenantcore-auth",
          aud: "tenantcore-api",
          sub: user.id,
          exp: now + 900,
          iat: now,
          nbf: now,
          jti: SecureRandom.uuid,
          tenant_id: user.tenant_id,
          tenant_role: tenant_role,
          groups: groups,
          permissions: permissions,
          acl_ver: user.acl_ver
        }

        JWT.encode(payload, @signing_key, "RS256")
        # test mode: check ENV["TENANTCORE_TEST_SECRET"] → "HS256" + hmac key
      end
    end

  lib/tenantcore_auth/auth/jwt_plug.ex:
    defmodule TenantcoreAuth.Auth.JwtPlug do
      import Plug.Conn
      alias TenantcoreAuth.Auth.{TokenService, AuthContext}
      alias TenantcoreAuth.Domain.InMemoryStore

      def init(opts), do: opts

      def call(conn, _opts) do
        with {:ok, token} <- extract_token(conn),
             {:ok, claims} <- TokenService.verify_token(token),
             {:ok, user} <- InMemoryStore.get_user_by_id(claims["sub"]),
             :ok <- verify_acl_ver(claims, user),
             :ok <- verify_membership(claims, user) do
          auth = build_auth_context(claims, user)
          assign(conn, :auth, auth)
        else
          {:error, :missing_token} ->
            conn  # let route plug handle missing auth
          {:error, reason} ->
            conn
            |> send_resp(401, Jason.encode!(%{error: "Unauthorized", reason: reason}))
            |> halt()
        end
      end

      # private helpers
    end

  lib/tenantcore_auth/auth/rbac_plug.ex:
    defmodule TenantcoreAuth.Auth.RbacPlug do
      import Plug.Conn

      def init(permission), do: permission

      def call(conn, permission) do
        case conn.assigns[:auth] do
          nil ->
            conn |> send_resp(401, Jason.encode!(%{error: "Unauthorized"})) |> halt()
          auth ->
            if AuthContext.has_permission?(auth, permission) do
              conn
            else
              conn |> send_resp(403, Jason.encode!(%{error: "Forbidden"})) |> halt()
            end
        end
      end
    end
