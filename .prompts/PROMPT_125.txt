BACKGROUND
================================================================================

SCALA: The Scala faker ecosystem is thinner than Python, JS, or Java.
       The most honest approach is: use Datafaker (via Scala/JVM interop) as the
       primary provider, and show idiomatic Scala patterns in the orchestration
       layer — lazy evaluation, case classes, functional pipelines, ZIO or Cats
       Effect if a streaming pipeline adds clarity, otherwise just LazyList.
       Do NOT pretend there is a thriving Scala-native faker ecosystem.
       State this clearly in the README and in the stack file.

ELIXIR: Elixir's functional nature means the generation pipeline is naturally
        expressed as a pipeline of Enum.map/Enum.flat_map/Stream transforms.
        The Faker library (elixirs/faker) provides atomic value generation.
        ExMachina provides factory-style generation with optional Ecto integration.
        The natural delivery mechanism in Elixir is a Mix task (mix generate).
        This arc's example must use Mix tasks, not a separate CLI binary.

B-3. lib/tenant_core/distributions.ex

Elixir weighted random using :rand (seeded Erlang RNG):

  defmodule TenantCore.Distributions do
    def weighted_plan(seed_state) do
      choices = [{"free", 50}, {"pro", 35}, {"enterprise", 15}]
      weighted_pick(choices, seed_state)
    end

    defp weighted_pick(weighted_list, _seed_state) do
      total = Enum.sum(Enum.map(weighted_list, fn {_, w} -> w end))
      target = :rand.uniform(total)
      # walk the list until cumulative weight >= target
      Enum.reduce_while(weighted_list, 0, fn {value, weight}, acc ->
        new_acc = acc + weight
        if new_acc >= target, do: {:halt, value}, else: {:cont, new_acc}
      end)
    end
  end

Seeding in Elixir:
  :rand.seed(:exsss, {42, 0, 0})   # seed the process-level RNG

Document: Elixir's :rand module seeds the process-level RNG. All :rand calls
in a single process share this state. Seed once at pipeline startup.
The Faker library also uses :rand internally — seeding :rand affects Faker's output.

B-4. lib/tenant_core/pipeline.ex

Functional pipeline using Enum and Map:

  defmodule TenantCore.Pipeline do
    alias TenantCore.{Domain, Distributions, Pools}

    def generate(%TenantCore.Profile{} = profile) do
      :rand.seed(:exsss, {profile.seed, 0, 0})

      orgs = build_orgs(profile.num_orgs)
      org_ids = Enum.map(orgs, & &1.id)

      users = build_users(profile.num_users)
      user_ids = Enum.map(users, & &1.id)

      {memberships, org_member_map} = build_memberships(orgs, user_ids)
      {projects, project_org_map} = build_projects(orgs, org_member_map)
      audit_events = build_audit_events(projects, org_member_map, project_org_map)
      api_keys = build_api_keys(orgs, org_member_map)
      invitations = build_invitations(orgs, org_member_map, users)

      dataset = %{
        organizations: orgs,
        users: users,
        memberships: memberships,
        projects: projects,
        audit_events: audit_events,
        api_keys: api_keys,
        invitations: invitations
      }

      report = TenantCore.Validate.check(dataset)
      unless report.ok, do: raise("Validation failed: #{inspect(report.violations)}")
      Map.put(dataset, :report, report)
    end

    defp build_orgs(count) do
      Enum.map(1..count, fn _ ->
        %TenantCore.Organization{
          id: UUID.uuid4(),
          name: Faker.Company.name(),
          slug: Faker.Internet.slug(),
          plan: Distributions.weighted_plan(nil),
          region: Distributions.weighted_region(nil),
          created_at: Faker.DateTime.backward(365 * 3) |> DateTime.to_iso8601(),
          owner_email: Faker.Internet.email()
        }
      end)
    end
    # ... similar for users, memberships, projects, events, api_keys, invitations
  end

Show the Enum.map pipeline idiom throughout. Use Stream for large profiles
(document this in the README):
  "For medium/large profiles, replace Enum.map with Stream.map and add a
   chunked write step using Stream.chunk_every/2."

  defmodule TenantCore.SmokeTest do
    use ExUnit.Case

    test "smoke profile generates 3 organizations" do
      dataset = TenantCore.Pipeline.generate(TenantCore.Profile.smoke())
      assert length(dataset.organizations) == 3
    end

      dataset = TenantCore.Pipeline.generate(TenantCore.Profile.smoke())
      assert dataset.report.ok == true
      assert dataset.report.violations == []
    end

    test "smoke profile is reproducible" do
      d1 = TenantCore.Pipeline.generate(TenantCore.Profile.smoke())
      d2 = TenantCore.Pipeline.generate(TenantCore.Profile.smoke())
      assert d1.organizations == d2.organizations
    end

    test "smoke profile has at least one non-free organization (weighted distribution)" do
      dataset = TenantCore.Pipeline.generate(TenantCore.Profile.small())
      plans = Enum.map(dataset.organizations, & &1.plan)
      free_count = Enum.count(plans, &(&1 == "free"))
      # free should be roughly 50%; allow wider range for small N
      assert free_count < length(plans)
