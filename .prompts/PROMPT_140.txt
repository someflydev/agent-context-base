  ## Key Distinctions to Preserve

  A bulleted list of the distinctions most likely to be lost by future assistants:
    - The database is the source of truth. The token is a snapshot. Never
      trust token-only data for high-stakes operations without a DB check.
    - acl_ver is not optional. A token with a stale acl_ver must be rejected.
      Missing this check makes group/permission changes ineffective until expiry.
    - Permissions are named {service}:{resource}:{action}. Route URL strings
      are not the permission namespace.
    - Route metadata is the source of truth for auth requirements. Inline
      permission checks that bypass the registry create drift.
    - Tenant-admin composes from a platform-owned permission catalog. If the
      example allows tenant-admins to name arbitrary permissions, it is wrong.
    - /me's allowed_routes must be computed from the route registry filtered
      by effective permissions. A hardcoded list is a defect.
    - super_admin is a global role with no tenant_id. Any code path that
      puts super_admin logic inside the tenant-scoped check path is wrong.
    - The canonical examples use an in-memory store. Production systems
      replace this with a real database — and must still enforce all the
      same invariants.
    - Python's test mode uses HS256 via TENANTCORE_TEST_SECRET. All other
      languages have a parallel test-mode escape hatch documented in their
      README. These are teaching aids, not production patterns.

This is the cross-stack parity runner. It checks that all 8 language

  #!/usr/bin/env python3
  """
  JWT Auth arc parity check runner.

  Checks:
  1. All 8 implementation directories exist.
  2. CATALOG.md marks all 8 implementations as [x].
  3. parity-matrix.md has no [ ] (not started) cells.
  4. Fixture files (users, tenants, groups, permissions, memberships) exist
     and are valid JSON with minimum entry counts.
  5. Domain spec contains all required sections.
  6. All 8 README files exist and contain required sections.
  """

  import sys, json, pathlib, unittest

  REPO_ROOT = pathlib.Path(__file__).parent.parent.parent
  AUTH_DIR = REPO_ROOT / "examples" / "canonical-auth"
  FIXTURE_DIR = AUTH_DIR / "domain" / "fixtures"

  REQUIRED_IMPLEMENTATIONS = [
    "python", "typescript", "go", "rust", "java", "kotlin", "ruby", "elixir"
  ]

  REQUIRED_FIXTURE_FILES = [
    "users.json", "tenants.json", "groups.json",
    "permissions.json", "memberships.json"
  ]

  REQUIRED_FIXTURE_MINIMUMS = {
    "users.json": 6,
    "tenants.json": 2,
    "groups.json": 3,
    "permissions.json": 10,
    "memberships.json": 5
  }

  REQUIRED_SPEC_SECTIONS = [
    "## The Permission Catalog",
    "## JWT Claim Shape Standard",
    "## Canonical Flows",
    "## Route Metadata Specification",
    "## /me Response Shape",
    "## Fixture Contract"
  ]

  REQUIRED_README_SECTIONS = [
    "How to run",
    "How to test"
  ]

  class TestImplementationDirectories(unittest.TestCase):
    def test_all_implementation_dirs_exist(self):
      for lang in REQUIRED_IMPLEMENTATIONS:
        d = AUTH_DIR / lang
        self.assertTrue(d.exists(), f"Missing implementation directory: {d}")

  class TestCatalogCompleteness(unittest.TestCase):
    def test_catalog_exists(self):
      self.assertTrue((AUTH_DIR / "CATALOG.md").exists())

    def test_all_implementations_marked_complete(self):
      catalog = (AUTH_DIR / "CATALOG.md").read_text()
      display_names = {
        "python": "Python", "typescript": "TypeScript",
        "go": "Go", "rust": "Rust", "java": "Java",
        "kotlin": "Kotlin", "ruby": "Ruby", "elixir": "Elixir"
      }
      for lang, name in display_names.items():
        found_complete = False
        for line in catalog.splitlines():
          if name in line and "[x]" in line:
            found_complete = True
            break
        self.assertTrue(found_complete,
          f"{name} not marked [x] in CATALOG.md")

  class TestParityMatrixCompleteness(unittest.TestCase):
    def test_no_not_started_cells(self):
      matrix = (AUTH_DIR / "domain" / "parity-matrix.md").read_text()
      not_started = [line for line in matrix.splitlines()
                     if "| [ ]" in line or "|[ ]" in line]
      self.assertEqual(not_started, [],
        f"Parity matrix has incomplete rows:\n" + "\n".join(not_started))

  class TestFixtureFiles(unittest.TestCase):
    def test_fixture_files_exist_and_valid(self):
      for fname in REQUIRED_FIXTURE_FILES:
        path = FIXTURE_DIR / fname
        self.assertTrue(path.exists(), f"Missing fixture: {path}")
        data = json.loads(path.read_text())
        self.assertIsInstance(data, list, f"{fname} should be a JSON array")
        minimum = REQUIRED_FIXTURE_MINIMUMS.get(fname, 1)
        self.assertGreaterEqual(len(data), minimum,
          f"{fname}: expected >= {minimum} entries, got {len(data)}")

  class TestDomainSpecSections(unittest.TestCase):
    def test_spec_has_required_sections(self):
      spec = (AUTH_DIR / "domain" / "spec.md").read_text()
      for section in REQUIRED_SPEC_SECTIONS:
        self.assertIn(section, spec, f"Missing section in spec.md: {section}")

  class TestReadmeFiles(unittest.TestCase):
    def test_all_readmes_exist_with_required_sections(self):
      for lang in REQUIRED_IMPLEMENTATIONS:
        readme = AUTH_DIR / lang / "README.md"
        self.assertTrue(readme.exists(), f"Missing README: {readme}")
        text = readme.read_text()
        for section in REQUIRED_README_SECTIONS:
          self.assertIn(section, text,
            f"README for {lang} missing section: '{section}'")

  class TestArcDocumentation(unittest.TestCase):
    def test_arc_overview_exists(self):
      p = REPO_ROOT / "docs" / "jwt-auth-arc-overview.md"
      self.assertTrue(p.exists(), "Missing docs/jwt-auth-arc-overview.md")

    def test_arc_overview_has_implementation_matrix(self):
      text = (REPO_ROOT / "docs" / "jwt-auth-arc-overview.md").read_text()
      self.assertIn("Implementation Matrix", text)

    def test_arc_overview_has_all_eight_languages(self):
      text = (REPO_ROOT / "docs" / "jwt-auth-arc-overview.md").read_text()
      for lang in ["Python", "TypeScript", "Go", "Rust", "Java", "Kotlin", "Ruby", "Elixir"]:
        self.assertIn(lang, text, f"Arc overview missing language: {lang}")

  if __name__ == "__main__":
    result = unittest.main(exit=False, verbosity=2)
    sys.exit(0 if result.result.wasSuccessful() else 1)
